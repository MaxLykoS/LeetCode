class Solution {
public:
vector<int> sortArrayByParityII(vector<int>& A) {
         int odd = 1, pair = 0, len = A.size();
        int tmp = 0;
        vector<int> ans(len);
        for (int i = 0; i < len; i++) {
            if (A[i] & 1) {
                ans[odd] = A[i];
                odd += 2;
            } else {
                ans[pair] = A[i]; 
                pair += 2;
            }
        }
        return ans;
    }
};
