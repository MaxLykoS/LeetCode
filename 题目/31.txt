class Solution {
public:
void nextPermutation(vector<int>& nums) 
{
	int n = nums.size();
	int i, j;
	for (i = n - 2; i >= 0; --i) 
	{
		if (nums[i] < nums[i + 1])
			break;
	}
	//对应于两个特例，需要判断i是否越界
	if (i >= 0) 
	{
		for (j = n - 1; j > i; --j) 
			if (nums[j] > nums[i]) 
				break;
		swap(nums[i], nums[j]);
	}
	reverse(nums.begin() + i + 1, nums.end());
}
};
