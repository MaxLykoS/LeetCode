class Solution {
public:
int maxSubArray(vector<int>& nums) 
{
	int pre,cur,
		maxVal = nums[0],
		size=nums.size();
	if (size == 0)
		return 0;
	pre = nums[0];
	for (int i = 1;i < size;i++)
	{
		cur = max(pre + nums[i], nums[i]);
		if (cur > maxVal)
			maxVal = cur;
		pre = cur;
	}
	return maxVal;
}
};
