class Solution {
public:
void rotate(vector<int>& nums, int k) 
{
	int size = nums.size();
	//k = k % size;
	int count = 0;
	for (int start = 0; count < size; ++start)
	{
		int current = start;
		int prev = nums[start];
		do {
			int next = (current + k) % size;
			int temp = nums[next];
			nums[next] = prev;
			prev = temp;
			current = next;
			count++;
		} while (start != current);
	}
}
};
